SELECT
	ASTRTP, CHAR(ASTDAT) AS ASTDAT, ASCOMP, ASCUST, ltrim(rtrim(ASINV#)) AS ASINV#, ASDUAM, ASINAM, DIGITS(ASFSYR)||DIGITS(ASFSPR) AS PER, ASCOMP||ASARAC AS ACCOUNT, CHAR(ASPDAT) AS ASPDAT
FROM
	LGDAT.AROP AROP
	INNER JOIN LGDAT.CUST ON
		BVCUST = ASCUST
WHERE
	ASCOMP = '66' AND
	BVARCD = 'F' AND
	DIGITS(ASFSYR)||DIGITS(ASFSPR) <= '1402'
	
UNION ALL

SELECT
	ASTRTP, CHAR(ASTDAT) AS ASTDAT, ASCOMP, ASCUST, ltrim(rtrim(ASINV#)) AS ASINV#, ASDUAM, ASINAM, DIGITS(ASFSYR)||DIGITS(ASFSPR) AS PER, ASCOMP||ASARAC AS ACCOUNT, CHAR(ASPDAT) AS ASPDAT
FROM
	LGDAT.AROHT
	INNER JOIN LGDAT.CUST ON
		BVCUST = ASCUST
WHERE
	ASCOMP = '66' AND
	BVARCD = 'F' AND
	CHAR(ASPDAT) >= '2014-03-01'
