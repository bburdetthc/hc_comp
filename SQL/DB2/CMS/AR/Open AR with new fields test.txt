SELECT
	ASTRTP, CHAR(ASTDAT) AS ASTDAT, ASCOMP, ASCUST, 
	DIGITS(ASFSYR)||DIGITS(ASFSPR) AS PER, 
	LTRIM(RTRIM(ASINV#)) AS ASINV#, 
	ASINAM AS INVOICE, ASDIS,ASADJ, ASPAY, ASCPAM AS LASTPAY, ASDUAM AS BALANCE,
	ASCOMP||ASARAC AS ACCOUNT, CHAR(ASPDAT) AS ASPDAT
FROM
	LGDAT.AROP AROP
WHERE
	ASCOMP = '66' AND
	ASCUST = 'DILL0001' AND
	DIGITS(ASFSYR)||DIGITS(ASFSPR) <= '1312'

UNION ALL
