SELECT
	LBCOM#,LBVEN#, LBPO#, LBPOI#, LBRLG#, JSRLGI, LBSHP#, CHAR(LBRDAT) LBRDAT, LBQREC, LBPT#, LBDESC, LBVCH#, LBEXT,LBITYP, LBJOB#, YACOMP||KBGL# AS GL
FROM
	LGDAT.POMVAR
	LEFT OUTER JOIN LGDAT.PORCH ON
		JSRLG# = LBRLG# AND
		JSPO# = LBPO# AND
		JSLIN# = LBPOI#
	------NEED POI TO GRAB THE ACCOUNT NUMBER FOR POMVAR RECORDS----
	LEFT OUTER JOIN LGDAT.POI ON
		KBPO# = LBPO# AND
		KBITM# = LBPOI#
	------NEED POH TO GRAB THE PLANT ASSOCIATED WITH THE PO IN ORDER TO LINK TO THE COMPANY AND USE THAT TO ATTACH TO THE GL ON POI------
	LEFT OUTER JOIN LGDAT.POH ON
		KAPO# = KBPO#
	------NEED PLNT TO GET THE COMPANY ASSOCIATED WITH THE PLANT TO ADD TO THE GL ON POI
	LEFT OUTER JOIN LGDAT.PLNT ON
		YAPLNT = KAPLNT
		
WHERE
	DIGITS(LBFSYY)||DIGITS(LBFSPP) > '1402'
FETCH FIRST 10 ROWS ONLY
