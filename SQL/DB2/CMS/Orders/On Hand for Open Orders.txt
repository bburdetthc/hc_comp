SELECT
	OO.*, SUM(QTOH)
FROM
(
SELECT
	O.PLNT, 
	O.DDPART,
	SUM(O.QTY_I) QTY_I
	SUM(O.VALUE_USD) VALUE_USD,
	SUM(O.COST_USD) COST_USD,
	O.GLEC, 
	O.GLDC,
	O.MAJG,
	O.MING,
	O.MAJS,
	O.MINS
FROM
	FANALYSIS.VW_FFOPOR O
WHERE
	FGRP LIKE '41010%'
GROUP BY
	O.PLNT, 
	O.DDPART,
	O.GLEC, 
	O.GLDC,
	O.MAJG,
	O.MING,
	O.MAJS,
	O.MINS
) OO
LEFT OUTER JOIN FANALYSIS.VW_FFSTKB ON
	BXPART = DDPART AND
	PLANT = PLNT
FETCH FIRST 10 ROWS ONLY
