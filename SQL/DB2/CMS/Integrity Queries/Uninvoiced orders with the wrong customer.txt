SELECT
	YACOMP, DCORD#, DCBCUS, BVCOMP, DCSTAT, CHAR(DCODAT) DCODAT
FROM	
	LGDAT.OCRH O
	LEFT OUTER JOIN LGDAT.CUST ON
		BVCUST = DCBCUS
	LEFT OUTER JOIN LGDAT.PLNT ON
		YAPLNT = DCPLNT
WHERE
	DCINV# = 0 AND
	YACOMP<>BVCOMP AND
	BVCOMP NOT IN ('36','39') AND
	DCODAT >= '2013-01-01'
ORDER BY DCODAT DESC
