SELECT
	DIGITS(DHARYR)||DIGITS(DHARPR) AS PERD, DHINCR, DIGLCD, DDORD#, DDGLC, DIPART, DIINV#, DILIN#, DHRINV, DHBCS#, BVNAME, DHCOMP, BVCOMP, BVCLAS, DIEXTN
FROM
	LGDAT.OIH
	INNER JOIN LGDAT.OID ON
		DIINV# = DHINV#
	INNER JOIN LGDAT.CUST ON
		BVCUST = DHBCS#
	LEFT OUTER JOIN LGDAT.OCRI ON
		DDORD# = DIORD# AND
		DDITM# = DIITM#
	INNER JOIN QGPL.PTCODEFLAG ON
		INCR = DHINCR AND
		GLCD = DIGLCD
WHERE
	DHARYR = 14 AND
	DHARPR = 1
FETCH FIRST 100 ROWS ONLY
