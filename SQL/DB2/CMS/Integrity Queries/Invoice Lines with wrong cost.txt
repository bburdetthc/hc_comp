SELECT
	DHPLNT, DHARYR||DIGITS(DHARPR) AS PERD, DHINCR, DIINV#, DILIN#, DIPART, DIQTSH, DIUNIT, DICTEX, DIEXT, V6UNTI, IHCNV2
FROM
	LGDAT.OID
	INNER JOIN LGDAT.OIH ON
		DHINV# = DIINV#
	INNER JOIN LGDAT.STKA ON
		DIPART = V6PART AND
		DHPLNT = V6PLNT
	LEFT OUTER JOIN LGDAT.PUNIT ON
		IHPART = DIPART AND
		IHUNT1 = DIUNIT AND
		IHUNT2 = V6UNTI
	LEFT OUTER JOIN LGDAT.ICSTP ON
		CHPART = DIPART AND
		CHPLNT = DHPLNT
	LEFT OUTER JOIN LGDAT.ICSTM ON
		CGPART = DIPART AND
		CGPLNT = DHPLNT
	LEFT OUTER JOIN LGDAT.ICSTR ON
		Y0PART = DIPART AND
		Y0PLNT = DHPLNT
WHERE
	DHARYR = 14 AND
	DIUNIT <> V6UNTI
FETCH FIRST 100 ROWS ONLY
