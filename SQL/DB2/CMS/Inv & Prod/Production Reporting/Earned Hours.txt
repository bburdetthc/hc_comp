SELECT
	IXFSYY||DIGITS(IXFSPP) PERD, CHAR(IXDATE) IXDATE, IXDEPT, NWLABL, IXWRKO, IXPART, IXSEQ#, IXRSRC, IXMEN#/IXMCH#, N.AORESC, N.AOEFF, 
	N.AORUNS, ROUND(N.AO#MEN/N.AO#MCH,4) AS CREW, 
	CASE ABLABR WHEN 0 THEN AASTDR ELSE ABLABR END LRATE, 
	SUM(IXQTY) IXQTY, SUM(IXSCRP) IXSCRP,
	SUM((IXQTY+IXSCRP)/AORUNS*N.AO#MEN/N.AO#MCH)AS STDLAB,
	SUM((IXQTY+IXSCRP)/AORUNS) AS STDMCH,
	SUM(IXHRS*IXMEN#/IXMCH#) ACTLAB,
	SUM(IXHRS) ACTMCH
FROM
	LGDAT.PRTRAN
	LEFT OUTER JOIN LGDAT.METHDR N ON
		N.AOPART = IXPART AND
		N.AOPLNT = SUBSTR(IXDEPT,1,3) AND
		N.AOSEQ# = 10
	LEFT OUTER JOIN LGDAT.RPRH ON
		IXBTCH = NWBTID
	LEFT OUTER JOIN LGDAT.RESRE ON
		ABRESC = AORESC AND
		ABDEPT = AODEPT
	LEFT OUTER JOIN LGDAT.DEPTS ON
		AADEPT = AODEPT
WHERE
	IXFSYY = 14 AND
	IXFSPP = 2 AND
	SUBSTR(IXDEPT,1,3) = '152' AND
	IXPART <> ''
GROUP BY
	IXFSYY||DIGITS(IXFSPP) , CHAR(IXDATE) , IXDEPT, NWLABL, IXWRKO, IXPART, IXSEQ#, IXRSRC,IXMEN#/IXMCH#,  N.AORESC, N.AOEFF, 
	N.AORUNS, ROUND(N.AO#MEN/N.AO#MCH,4),CASE ABLABR WHEN 0 THEN AASTDR ELSE ABLABR END
FETCH FIRST 10 ROWS ONLY
