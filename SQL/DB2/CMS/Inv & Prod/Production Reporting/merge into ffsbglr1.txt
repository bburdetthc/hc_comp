MERGE INTO
	QGPL.FFSBGLR1 F
USING
	QGPL.FFSBGLWF W ON
		W.PERD = F.PERD AND
		W.MODULE = F.MODULE AND
		W.CUSMOD = F.CUSMOD AND
		W.ACCT = F.ACCT AND
		W.BATCH = F.BATCH AND
		W.PDATE = F.PDATE AND
		W.PROJ = F.PROJ AND
		W.CUSKEY1 = F.CUSKEY1 AND
		W.CUSKEY2 = F.CUSKEY2 AND
		W.CUSKEY3 = F.CUSKEY3 AND
		W.CUSKEY4 = F.CUSKEY4 AND
		W.CUSVEND = F.CUSVEND AND
		W.CUSCUST = F.CUSCUST AND
		W.RECID = F.RECID
WHEN NOT MATCHED THEN 
	INSERT (	F.MODULE, F.BATCH, F.PERD, F.TDATE, F.PDATE, 
			F.ACCT, F.AMT, F.PROJ, F.USRN, F.REV, F.CUSMOD, 
			F.CUSKEY1, F.CUSKEY1D, F.CUSKEY2, F.CUSKEY2D, 
			F.CUSKEY3, F.CUSKEY3D, F.CUSKEY4, F.CUSKEY4D,
			F.CUSVEND, F.CUSCUST, F.RECID
		)
	VALUES	(	W.MODULE, W.BATCH, W.PERD, W.TDATE, W.PDATE, 
			W.ACCT, W.AMT, W.PROJ, W.USRN, W.REV, W.CUSMOD, 
			W.CUSKEY1, W.CUSKEY1D, W.CUSKEY2, W.CUSKEY2D, 
			W.CUSKEY3, W.CUSKEY3D, W.CUSKEY4, W.CUSKEY4D,
			W.CUSVEND, W.CUSCUST, W.RECID
		)
