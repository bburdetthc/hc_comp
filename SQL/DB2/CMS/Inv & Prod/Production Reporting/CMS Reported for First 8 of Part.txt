SELECT
	
	IXPART, IXDEPT, IXRSRC, DIGITS(IXFSYY)||DIGITS(IXFSPP) AS PER, 
	SUM(IXQTY) AS GOOD, SUM(IXSCRP) AS SCRAP, SUM(IXHRS) AS ACTUAL, 
	AORUNS, AO#MEN, AO#MCH

FROM
	
	LGDAT.PRTRAN
	
	LEFT OUTER JOIN LGDAT.METHDR ON
		
		AOPART = IXPART AND
		
		SUBSTR(AODEPT,1,3) = SUBSTR(IXDEPT,1,3) AND
		
		AOSEQ# = 10

WHERE
	
	IXPART LIKE 'HZP3E100%' AND
	
	IXFSYY >= 12

GROUP BY
	
	IXPART, IXDEPT, IXRSRC, DIGITS(IXFSYY)||DIGITS(IXFSPP), AORUNS, AO#MEN, AO#MCH
