SELECT
	V6PART, V6PLNT, MOLD, V6RPLN, V6STAT, 
	N.AORESC, N.AOEFF, N.AORUNS, ROUND(N.AO#MEN/N.AO#MCH,4) AS CREW,
	O.AORESC, O.AOEFF, O.AORUNS, ROUND(O.AO#MEN/O.AO#MCH,4) AS CREW
FROM
	LGDAT.STKA
	INNER JOIN QTEMP.MLIST ON
		MOLD = SUBSTR(V6PART,1,8)
	LEFT OUTER JOIN LGDAT.METHDR N ON
		N.AOPART = V6PART AND
		N.AOPLNT = V6PLNT AND
		N.AOSEQ# = 10
	LEFT OUTER JOIN LGDAT.METHDR O ON
		O.AOPART = V6PART AND
		O.AOPLNT = '112' AND
		O.AOSEQ# = 10
WHERE
	V6PLNT = '152' AND
	V6RPLN = '1'
