SELECT
	CHAR(TDT) RDAT, PART, PLNT, 
	YACOMP||DIGITS(Y1INVA) ACCT, 
	SUBSTR(YACOMP||DIGITS(Y1INVA),7,4) PRIME,
	COALESCE(AWGLDC, AVGLCD) GLCD,
	COALESCE(AWGLED, AVGLED) GLED,
	COALESCE(AWMAJG,AVMAJG)||' - '||RTRIM(BQDES) MAJG,
	COALESCE(AWMING, AVMING)||' - '||RTRIM(BRDES)  MING,
	V6STAT, V6RPLN,
	FCOST, TCOST, TQTY, 
	F.JHTOTN*TQTY 		FTOT, 
	T.JHTOTN*TQTY 		TTOT, 
	(T.JHTOTN-F.JHTOTN)*TQTY 	CTOT,
	(F.JHMATN+F.JHTOTN-F.JHMATN-F.JHLABN-F.JHBURN-F.JHOTHN-F.JHSHCN-F.JHAVDN-F.JHCRCN)*TQTY						FMAT, 
	(T.JHMATN+T.JHTOTN-T.JHMATN-T.JHLABN-T.JHBURN-T.JHOTHN-T.JHSHCN-T.JHAVDN-T.JHCRCN)*TQTY						TMAT,
	(T.JHMATN-F.JHMATN+(T.JHTOTN-T.JHMATN-T.JHLABN-T.JHBURN-T.JHOTHN-T.JHSHCN-T.JHAVDN-T.JHCRCN)-(F.JHTOTN-F.JHMATN-F.JHLABN-F.JHBURN-F.JHOTHN-F.JHSHCN-F.JHAVDN-F.JHCRCN))*TQTY 	CMAT,
	F.JHSHCN*TQTY 		FSHH, 
	T.JHSHCN*TQTY 		TSHH, 
	(T.JHSHCN-F.JHSHCN)*TQTY 	CSHH,
	F.JHAVDN*TQTY 		FDUT, 
	T.JHAVDN*TQTY 		TDUT, 
	(T.JHAVDN-F.JHAVDN)*TQTY 	CDUT,
	F.JHCRCN*TQTY 		FCRN, 
	T.JHCRCN*TQTY 		TCRN, 
	(T.JHCRCN-F.JHCRCN)*TQTY	CCRN,
	F.JHLABN*TQTY 		FLAB, 
	T.JHLABN*TQTY 		TLAB, 
	(T.JHLABN-F.JHLABN)*TQTY 	CLAB,
	F.JHBRVN*TQTY 		FVAR,
	T.JHBRVN*TQTY 		TVAR,  
	(T.JHBRVN-F.JHBRVN)*TQTY 	CVAR,
	F.JHBRFN*TQTY 		FFIX, 
	T.JHBRFN*TQTY 		TFIX, 
	(T.JHBRFN-F.JHBRFN)*TQTY 	CFIX,
	F.JHOTHN*TQTY 		FOTH,	
	T.JHOTHN*TQTY 		TOTH,
	(T.JHOTHN-F.JHOTHN)*TQTY	COTH
FROM
	QGPL.FFCOSTEFFD C
	INNER JOIN LGDAT.ICSTT F ON
		F.JHRCID = FRI
	INNER JOIN LGDAT.ICSTT T ON
		T.JHRCID = TRI
	LEFT OUTER JOIN LGDAT.STKA ON
		V6PART = PART AND
		V6PLNT = PLNT
	LEFT OUTER JOIN LGDAT.STKMM ON
		AVPART = PART
	LEFT OUTER JOIN LGDAT.STKMP ON
		AWPART = PART
	LEFT OUTER JOIN LGDAT.PLNT ON
		YAPLNT = PLNT
	LEFT OUTER JOIN LGDAT.GLIE ON
		Y1PLNT = PLNT AND
		Y1GLEC = COALESCE(AVGLED, AWGLED)
	LEFT OUTER JOIN LGDAT.MAJG ON
		BQGRP = COALESCE(AWMAJG, AVMAJG)
	LEFT OUTER JOIN LGDAT.MMGP ON
		BRGRP = BQGRP AND
		BRMGRP = COALESCE(AVMING, AWMING)
WHERE
	FCOST-TCOST<>0 AND
	TQTY <> 0 AND
	PART = 'RZD102H0000C018'
FETCH FIRST 100 ROWS ONLY
