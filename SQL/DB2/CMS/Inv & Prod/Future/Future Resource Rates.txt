SELECT
	ABPLNT, ABDEPT, ABRESC, ABDES, 
	CASE R9LABR WHEN 0 THEN R8STDR ELSE R9LABR END,
	CASE R9BRDR WHEN 0 THEN R8BRDR ELSE R9BRDR END, 
	CASE R9VBRD WHEN 0 THEN R8VBRD ELSE R9VBRD END
FROM
	LGDAT.RESRE
	LEFT OUTER JOIN LGDAT.FRESRE ON
		ABDEPT = R9DEPT AND
		ABRESC = R9RESC
	LEFT OUTER JOIN LGDAT.FDEPTS DEPTS ON
		R9DEPT = R8DEPT
WHERE
	ABPLNT NOT IN ('151','111')
