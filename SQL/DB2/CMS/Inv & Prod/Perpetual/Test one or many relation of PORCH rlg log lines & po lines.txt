SELECT 
	'POLIN-RLGLIN' UNQ, COUNT(*)
FROM
(
SELECT 
	JSPO#, JSLIN#, JSRLG#, JSRLGI
FROM
	LGDAT.PORCH
WHERE
	JSRDAT >= '2014-01-01'
) X

UNION ALL

SELECT 
	'POLIN' AS UNQ, COUNT(*)
FROM
(
SELECT 
	JSPO#, JSLIN#
FROM
	LGDAT.PORCH
WHERE
	JSRDAT >= '2014-01-01'
) X

UNION ALL

SELECT 
	'RLGLIN' UNQ, COUNT(*)
FROM
(
SELECT 
	JSRLG#, JSRLGI
FROM
	LGDAT.PORCH
WHERE
	JSRDAT >= '2014-01-01'
) X
