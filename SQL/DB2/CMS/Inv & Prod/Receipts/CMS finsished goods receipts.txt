SELECT 
	'IV' AS GL, DKACC#, DKFSPR, DKSRCE, CASE DKQUAL WHEN '' THEN UPPER(SUBSTR(DKADDD,1,4)) ELSE DKQUAL END AS DKQUAL, Sum(DKAMT)
FROM 
	LGDAT.GLSBIV GLSBIV
WHERE
	DKACC# = '935200122000' AND 
	DKFSYY =12 AND 
	DKFSYR = 20 AND
	DKSRCE = 'IC' AND
	DKQUAL = 'RL'
GROUP BY 
	DKACC#, DKFSPR, DKSRCE, CASE DKQUAL WHEN '' THEN UPPER(SUBSTR(DKADDD,1,4)) ELSE DKQUAL END

UNION ALL

SELECT 
	'PD' AS GL, DKACC#, DKFSPR, DKSRCE, CASE DKQUAL WHEN '' THEN UPPER(SUBSTR(DKADDD,1,4)) ELSE DKQUAL END AS DKQUAL, Sum(DKAMT)
FROM 
	LGDAT.GLSBIV GLSBIV
WHERE
	DKACC# = '935200122000' AND 
	DKFSYY =12 AND 
	DKFSYR = 20 AND
	DKSRCE = 'PD' AND
	DKQUAL = 'WO' AND
	SUBSTR(DKREFD,1,7) = 'CP- PO#'
GROUP BY 
	DKACC#, DKFSPR, DKSRCE, CASE DKQUAL WHEN '' THEN UPPER(SUBSTR(DKADDD,1,4)) ELSE DKQUAL END
