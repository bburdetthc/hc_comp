INSERT INTO QGPL.FFBPP
SELECT
	PART, PERD, SUM(QTY), SUM(FPV), SUM(ABS)
FROM
	(
		SELECT
			PARENTPART PART, SUBSTR(DIGITS(YEARF),3,2)||DIGITS(MONTHF) PERD, 0 QTY, FPVVALUE FPV, 0 ABS
		FROM
			LGPGM.BI007FPV
		WHERE
			PLANTF IN ('152','MID') AND
			SUBSTR(GLEXPENSEF,1,1) IN ('1','2')
	
		UNION ALL
	
		SELECT
			PART, SUBSTR(DIGITS(YEAR),3,2)||DIGITS(MONTH) PERD, BUILDQTY QTY, 0 FPV, OHABS+LABABS ABS
		FROM
			LGPGM.BI006BLD
		WHERE
			PLANT IN ('152','MID') 
	) X
GROUP BY
	PART, PERD
