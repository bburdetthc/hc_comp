SELECT
	X.*, 
	1/FLOAT(FACTOR) Q_PALLET,
	(INVOICE_USD-COST_USD)/QTY_SHIP MARGD_PC,
	((INVOICE_USD-COST_USD)/QTY_SHIP )*(1/FLOAT(FACTOR) ) MARGPP,
	((INVOICE_USD-COST_USD)/QTY_SHIP )*(1/FLOAT(FACTOR) )*INVOICE_USD WAMARGPP

FROM
	QGPL.VW_FFINVD X
	LEFT OUTER JOIN LGDAT.STKA A ON
		V6PART = RTRIM(SUBSTR(PARTDESCR,1,20)) AND
		V6PLNT = INV_PLNT
	LEFT OUTER JOIN FANALYSIS.VW_FFUNIT P ON
		IHPART = RTRIM(SUBSTR(PARTDESCR,1,20)) AND
		IHUNT1 = V6UNTI AND
		IHUNT2 = 'PLT'
WHERE
	FISC_PERD >= '1503'
FETCH FIRST 10 ROWS ONLY
