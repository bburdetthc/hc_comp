SELECT DISTINCT
	UBCOMP,
	UBBTCH,
	UBPSBT,
	UBSTAT, 
	UCSTAT,
	DKPOST,
	UCFSYR||DIGITS(UCFSPR) AS JRNLPER,
	DKFSYY||DIGITS(DKFSPR) AS POSTPER,
	UBDESC,
	UBCRBY AS ENTERED_BY,
	UBRVBY AS REVISED_BY,
	DKFUT4 AS POSTED_BY,
	UBCDAT,
	DKPDAT,
	UCJTOT
FROM
	LGDAT.JRNB
	LEFT OUTER JOIN LGDAT.JRNA ON
		UCBTCH = UBBTCH AND
		UCCOMP =  UBCOMP
	LEFT OUTER JOIN LGDAT.GTRAN ON
		DKBTC# = UBPSBT
WHERE	
	UCFSYR = 13 AND
	UCFSPR = 7
