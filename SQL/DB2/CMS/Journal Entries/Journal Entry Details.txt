SELECT 
	DKACC#, DKAMT, AZTITL, RTRIM(HYPOLD) AS HYP, DKREF#, RTRIM(DKADDD) AS DKADDD, DKREFD, 
	UBCRBY, CHAR(UBCDAT) AS UBCDAT, UBRVBY, CHAR(UBRDAT) AS UBRDAT, DKRCID,
	int(ROW_NUMBER() OVER ()) AS ROWNUMBER
	
FROM 
	LGDAT.GTRAN G
	INNER JOIN LGDAT.MAST M ON
		AZCOMP||DIGITS(AZGL#1)||DIGITS(AZGL#2) = DKACC#
	LEFT OUTER JOIN LGPGM.CMSTOHYP X ON
		CMSGL = DKACC#
	LEFT OUTER JOIN LGDAT.JRNA A ON
		UBCOMP = SUBSTR(DKACC#,1,2) AND
		DIGITS(UBBTCH) = RTRIM(DKKEYN)
	LEFT OUTER JOIN LGDAT.JRNB B ON
		UBCOMP = UCCOMP AND
		UBBTCH = UCBTCH
WHERE
	DKFSPR = 12 AND
	DKFSYY = 13 AND
	DKREF# = '000850-078'
ORDER BY DKRCID
