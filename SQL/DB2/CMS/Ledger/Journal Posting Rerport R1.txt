SELECT
	UBCOMP,
	UBBTCH,
	UCJRNL,
	UBPSBT,
	UBSTAT, 
	UCSTAT,
	UCFSYR||DIGITS(UCFSPR) AS JRNLPER,
	UBDESC,
	UBCRBY AS ENTERED_BY,
	UBRVBY AS REVISED_BY,
	CHAR(UCCDAT) JCDAT,
	CHAR(UCTDAT) JTDAT,
	CHAR(UCRDAT) JRDAT,
	CHAR(UBCDAT) BCDAT,
	UCJTOT
FROM
	LGDAT.JRNB
	LEFT OUTER JOIN LGDAT.JRNA ON
		UCBTCH = UBBTCH AND
		UCCOMP =  UBCOMP
WHERE	
	UCFSYR = 14
FETCH FIRST 100 ROWS ONLY
