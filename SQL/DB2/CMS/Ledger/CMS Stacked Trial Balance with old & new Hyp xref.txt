SELECT 
	DIGITS(AZCOMP) AS AZOMP, SUBSTR(DIGITS(AZGL#2),1,4) AS OBJ, AJ4COMP||DIGITS(AJ4GL#1)||DIGITS(AJ4GL#2) AS ACC, MAST.AZSTAT,
	MAST.AZTITL, MAST.AZTTL2, 
	MAST.AZFUT2, MAST.AZGROP, FGRP.BQ1LVL, 
	LTRIM(RTRIM(MAST.AZGROP))||' - '||LTRIM(RTRIM(FGRP.BQ1TITL)),  AZFUT3 AS CATCODE, D35DES1, HYPNEW AS HYPACCT,
	SUBSTR(AJ4CCYY,3,2)||'00' AS PERIOD,
	GLMT.AJ4OB01 AS AMOUNT
FROM 
	LGDAT.GLMT GLMT
	LEFT OUTER JOIN LGDAT.MAST MAST ON
		GLMT.AJ4COMP = MAST.AZCOMP AND 
		GLMT.AJ4GL#1 = MAST.AZGL#1 AND 
		GLMT.AJ4GL#2 = MAST.AZGL#2
	LEFT OUTER JOIN LGDAT.FGRP FGRP ON
		MAST.AZGROP = FGRP.BQ1GRP
	LEFT OUTER JOIN LGDAT.GGTP ON
		MAST.AZFUT3 = D35GCDE
	LEFT OUTER JOIN LGPGM.CMSTOHYP H ON
		CMSGL = AJ4COMP||DIGITS(AJ4GL#1)||DIGITS(AJ4GL#2)
WHERE 
	GLMT.AJ4CCYY = 2013 AND
	GLMT.AJ4OB01 <> 0

UNION ALL
