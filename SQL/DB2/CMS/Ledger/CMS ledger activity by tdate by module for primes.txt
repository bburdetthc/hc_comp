SELECT
	CHAR(DKTDAT) AS DKTDAT, DKACC#, SUM(DKAMT), DKSRCE||DKQUAL
FROM
	LGDAT.GLSBIV
WHERE
	SUBSTR(DKACC#,7,4) IN ('1200','1220') AND
	DKFSYY = 13 AND
	DKFSPR = 12
GROUP BY
	CHAR(DKTDAT),  DKACC#, DKSRCE||DKQUAL

UNION ALL
