SELECT
	UCFSYR||DIGITS(UCFSPR) AS FSPR,
	UBCOMP COMP,
	UBBTCH BATCH,
	UCJRNL JRNL#,
	UBPSBT GL_BATCH,
	UBSTAT POST_STAT, 
	UCREVE REVERSING,
	UBDESC DESCRIPTION,
	UBCRBY AS ENTERED_BY,
	UBRVBY AS REVISED_BY,
	CHAR(UCCDAT) JCDAT,
	CHAR(UCTDAT) JTDAT,
	CHAR(UCRDAT) JRDAT,
	CHAR(UBCDAT) BCDAT,
	UCJTOT
FROM
	LGDAT.JRNA
	INNER JOIN LGDAT.JRNB ON
		UCBTCH = UBBTCH AND
		UCCOMP =  UBCOMP
WHERE	
	UCFSYR||DIGITS(UCFSPR) = '1509'
	
FETCH FIRST 100 ROWS ONLY
