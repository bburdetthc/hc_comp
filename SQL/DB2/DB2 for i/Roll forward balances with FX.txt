SELECT 
	T.*, P.PRIORP
FROM 
	FANALYSIS.VW_FFTB T
	LEFT OUTER JOIN FANALYSIS.VW_PRFSPR P ON
		T.COMP = P.COMP AND
		T.FSPR = P.CURRP
	LEFT OUTER JOIN FANALYSIS.FFCRET C ON
		
WHERE 
	T.FGRP LIKE '1102010%' AND 
	T.FSPR >= '1502' AND 
	T.COMP >= '20' AND 
	SUBSTR(T.ACCOUNT,11,2) <> '00' 
ORDER BY 
	T.FSPR ASC
FETCH FIRST 10 ROWS ONLY
