SELECT 
	G.*, A.ASINAM, ASINV#
FROM 
	QGPL.FFSBGLR1 G
	LEFT OUTER JOIN LGDAT.AROHT A ON
		ASCUST = CUSCUST AND
		ASCOMP = SUBSTR(ACCT,1,2) AND
		RTRIM(CUSKEY2) = ASTRDS AND
		ASBCHQ = RTRIM(CUSKEY1)
WHERE 
	CUSCUST = 'MENA0007' AND 
	CUSMOD = 'DEBIT NOTE' AND
	PERD = '1403' AND
	MODULE = 'AR  '
FETCH FIRST 100 ROWS ONLY
