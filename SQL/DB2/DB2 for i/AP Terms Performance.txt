SELECT
	FHCOM#, FHBANK, FHVEN#, BTNAME, FHMDTE, FHCURR, FHGRS, FHDIS, FHPTD, FHCBAL, 
	FHSTSC, FHSTSK, 
	DAYS(FHDDAT) - DAYS(FHIDAT) DUEDAYS_G,
	DAYS(FHSDAT) - DAYS(FHIDAT) DISCDAYS_G,
	ROUND(FHDIS/FHGRS,3) DISCP_G,
	BTTRMC, DESCR,
	PAYDAYS DUEDAYS_S,
	DISCDAYS DISCDAYS_S,
	DISCP DISCP_S,
	DAYS(FHCDAT) - DAYS(FHIDAT) DAYS_TAKEN, 
	CASE FHCBAL WHEN 0 THEN FHGRS-FHPTD ELSE 0 END DISC_TAKE
FROM
	LGDAT.OPEN O
	LEFT OUTER JOIN LGDAT.VEND ON
		BTVEND = FHVEN#
	LEFT OUTER JOIN FANALYSIS.VW_FFTMCD ON
		TERM = BTTRMC
WHERE
	FHMDTE >= '201501'
